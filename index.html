var periodicTable = [ ['H', 'He'], ['Li', 'Be', 'B', 'C', 'N', 'O', 'F', 'Ne'], // ... rest der Perioden ]; function createTable(periodicTable) { var table = document.getElementById('periodic-table'); periodicTable.forEach(function(row) { var rowElement = document.createElement('div'); rowElement.className = 'row'; row.forEach(function(element) { var elementElement = document.createElement('div'); elementElement.className = 'element'; elementElement.textContent = element; rowElement.appendChild(elementElement); }); table.appendChild(rowElement); }); } createTable(periodicTable);
